------------------------------filtrare  i clienti maggiori di 30 anni

SELECT * 
FROM clienti
WHERE eta > 30;





------------------------------filtrare per nome = valore
SELECT * 
FROM clienti
WHERE nome = 'Giovanni';





------------------------------filtrare per range di eta
SELECT * 
FROM clienti
WHERE eta > 36
AND eta < 46;





------------------------------Ordine decrescente dell eta
SELECT *
FROM clienti
ORDER BY eta DESC;




------------------------------Ordine crescente dell eta
SELECT *
FROM clienti
ORDER BY eta ASC;






------------------------------MODIFICA DELL email

------------------------------cancellare utente in base all id
DELETE FROM clienti
WHERE id = 1;





------------------------------funzione di COUNT
SELECT COUNT(*) AS totale_clienti
FROM clienti;








------------------------------CREO TABELLA PRODOTTI
CREATE TABLE prodotti(
  		
    	id INT AUTO_INCREMENT PRIMARY KEY,
    	nome VARCHAR(50),
    	prezzo DECIMAL(10,2)
    
);





------------------------------popolo tabella PRODOTTI

------------------------------funzione somma dei prezzi di tutti i prodotti
SELECT SUM(prezzo) AS totale
FROM prodotti;






------------------------------media dei prezzi dei prodotti
SELECT AVG(prezzo) AS media
FROM prodotti;





------------------------------trovare il prezzo piu alto dei prodotti
SELECT MAX(prezzo)
FROM prodotti;





------------------------------trovare il prezzo piu basso dei prodotti
SELECT MIN(prezzo)
FROM prodotti;
